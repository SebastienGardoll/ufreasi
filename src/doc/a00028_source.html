447  S2=0.;

452  }
453

457  lines[Id].intercept = moyY-(lines[Id].slope)*moyX;

461  SD2 = S2 ;
462  S2=0.;
463 
464  for(int i=0;i<x.size();i++)
465  {
466  S1=S1+(y[i]-moyY)*(y[i]-moyY);
467  double val = lines[Id].slope*x[i]+lines[Id].intercept;
468  S2=S2+(y[i]-val)*(y[i]-val);
